const int N = 1e5 + 10;
const int M = 2e5 + 20;
int n, m, k, idx;
int h[N], to[M*2], ne[M*2], d[N], dis[N];  // d[] åº¦æ•°


void add(int u, int v) {
    to[idx] = v, ne[idx] = h[u], h[u] = idx++;    
}

void init() {
    memset(h, -1, sizeof(h));
    memset(dis, -1, sizeof(dis));
    dis[1] = 0;
    cin >> n >> m >> k;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        add(u, v);
        add(v, u);
        d[u]++;
        d[v]++;
    }
}

void bfs() {
    queue<int> q;
    q.push(1);    
    while (!q.empty()) {
        int t = q.front();
        q.pop();
        for (int i = h[t]; ~i; i = ne[i]) {  // ~i ç­‰ä»·äº i != -1
            int v = to[i];
            if (dis[v] == -1) {
                dis[v] = dis[t] + 1;
                q.push(v);
            }
        }
    }
}

void solve() {
    bfs();
    LL ans = 1 + d[1] * k;  // è®°å¾—åŠ ä¸Šç»“ç‚¹ 1 æœ¬èº«

            ans += (LL)(k - dis[i]) * (LL)(d[i] - 2);
        }
    }  
    cout << ans << "\n";
}
