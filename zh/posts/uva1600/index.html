def read_lines(m):
    pic = []
    for _ in range(m):
        pic.append(list(map(int, input().split())))
    return pic

def check(x, y, m, n):
    if (x < 0 or y < 0 or x > m or y > n):
        return True
    return False

def bfs(x, y, m, n, k):
    global pic, vis
    dx = [0, 1, 0, -1]
    dy = [1, 0, -1, 0]
    q = [[x, y, 0, 0]]  # åˆ©ç”¨åˆ—è¡¨[x, y, step, cnt]å­˜å‚¨çŠ¶æ€ï¼Œqä¸ºé˜Ÿåˆ—

    return -1
